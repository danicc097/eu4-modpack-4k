make_vassal_to_personal_union = {
	category = influence

	alert_index = 5
	alert_tooltip = make_vassal_to_personal_union_tooltip

	require_acceptance = no

	is_visible = {
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = vassal
		}
	}

	is_allowed = {
		hidden_trigger = {
			FROM = {
				is_subject_of = ROOT
				is_subject_of_type = vassal
			}
		}
		has_regency = no
		marriage_with = FROM
		NOT = { truce_with = FROM }
		is_at_war = no
		FROM = {
			is_at_war = no
		}
		months_of_ruling = 24
		FROM = {
			months_of_ruling = 24
		}
		FROM = {
			NOT = { liberty_desire = 50 }
		}
		FROM = {
			years_in_vassalage_under = {
				who = ROOT
				years = 5
				all_subject_country = yes
			}
		}
		FROM = {
			has_opinion = {
				who = ROOT
				value = 190
			}
		}
	}

	on_accept = {
		create_subject = {
			subject = FROM
			subject_type = personal_union
		}
		add_stability = -1
		add_prestige = -25
		FROM = {
			add_opinion = {
				who = ROOT
				modifier = they_force_unioned_us
			}
			add_opinion = {
				who = ROOT
				modifier = claimed_throne_on_us
			}
		}
	}
}
