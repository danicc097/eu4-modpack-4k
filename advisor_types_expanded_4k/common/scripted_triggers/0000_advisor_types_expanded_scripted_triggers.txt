has_adm_advisor = {
	custom_trigger_tooltip = {
		tooltip = has_adm_advisor_tooltip
		OR = {
			advisor = treasurer
			advisor = philosopher
			advisor = artist
			advisor = theologian
			advisor = master_of_mint
			advisor = inquisitor
			advisor = natural_scientist
			# Advisor Types Expanded
			has_ate_advisor = { adm = yes skill = 1 }
			# Compatibility with Advisors Expanded
			has_advisors_expanded_advisor = { adm = yes skill = 1 }
			# Compatibility with Flavor Universalis
			has_flavor_universalis_advisor  = { adm = yes skill = 1 }
		}
	}
}
