						}
					}

						limit = {
							NOT = { country_or_non_sovereign_subject_holds = ROOT }
							NOT = { is_claim = ROOT }
							NOT = { is_core = ROOT }
						}
						save_event_target_as = claim_province
					}
				}
			}
		}

			}
