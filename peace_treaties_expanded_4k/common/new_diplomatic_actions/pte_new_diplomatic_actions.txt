pte_start_border_friction = {
	category = covert

	require_acceptance = no

	is_visible = {
		is_neighbor_of = FROM
		is_free_or_tributary_trigger = yes
		FROM = {
			is_free_or_tributary_trigger = yes
		}
		NOT = { overlord_of = FROM }
		NOT = { is_subject_of = FROM }
	}

	is_allowed = {
		hidden_trigger = {
			is_neighbor_of = FROM
			is_free_or_tributary_trigger = yes
			FROM = {
				is_free_or_tributary_trigger = yes
			}
			NOT = { overlord_of = FROM }
			NOT = { is_subject_of = FROM }
		}
		if = {
			limit = {
				has_global_flag = extended_timeline_mod
			}
			if = {
				limit = {
					started_in = 1400.1.1
				}
				dip_tech = 50
			}
			else = {
				dip_tech = 30
			}
		}
		else = {
			dip_tech = 5
		}
		NOT = { corruption = 100 }
		has_spy_network_in = {
			who = FROM
			value = 60
		}
		NOT = { war_with = FROM }
		NOT = { truce_with = FROM }
		NOT = { alliance_with = FROM }
		FROM = {
			NOT = { guaranteed_by = ROOT }
		}
		NOT = {
			has_casus_belli = {
				target = FROM
				type = cb_pte_border_friction
			}
		}
	}

	on_accept = {
		if = {
			limit = {
				is_neighbor_of = FROM
				is_free_or_tributary_trigger = yes
				FROM = {
					is_free_or_tributary_trigger = yes
				}
				NOT = { overlord_of = FROM }
				NOT = { is_subject_of = FROM }
				OR = {
					AND = {
						has_global_flag = extended_timeline_mod
						OR = {
							AND = {
								started_in = 1444.11.11
								dip_tech = 50
							}
							AND = {
								NOT = { started_in = 1444.11.11 }
								dip_tech = 30
							}
						}
					}
					AND = {
						NOT = { has_global_flag = extended_timeline_mod }
						dip_tech = 5
					}
				}
				NOT = { corruption = 100 }
				has_spy_network_in = {
					who = FROM
					value = 60
				}
				NOT = { war_with = FROM }
				NOT = { truce_with = FROM }
				NOT = { alliance_with = FROM }
				FROM = {
					NOT = { guaranteed_by = ROOT }
				}
				NOT = {
					has_casus_belli = {
						target = FROM
						type = cb_pte_border_friction
					}
				}
			}
			add_corruption = 2
			add_spy_network_in = {
				who = FROM
				value = -60
			}
			add_casus_belli = {
				target = FROM
				type = cb_pte_border_friction
			}
		}
	}

	ai_will_do = {
		NOT = { has_casus_belli_against = FROM }
		NOT = { claim = FROM }
		NOT = { core_claim = FROM }
		FROM = {
			num_of_owned_provinces_with = {
				value = 5
				any_neighbor_province = {
					owned_by = ROOT
				}
			}
		}
	}
}
