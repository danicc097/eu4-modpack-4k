#Custom House of Apena Sheep
country_event = {
	id = GE_Italian_Reforms_Events.4
	title = GE_Italian_Reforms_Events.4.T
	desc = GE_Italian_Reforms_Events.4.D
	picture = WESTERNISATION_eventPicture

	hidden = yes

	trigger = {
		OR = {
			AND = {
				has_government_attribute = enables_customs_house
				any_owned_province = {
					OR = {
						trade_goods = cloth
						trade_goods = livestock
						trade_goods = wool
					}
					NOT = {	has_province_modifier = GE_Customs_House }
				}
			}
			AND = {
				NOT = { has_government_attribute = enables_customs_house }
				any_owned_province = {
					has_province_modifier = GE_Customs_House
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	immediate = {
		if = {
			limit = {
				NOT = { has_government_attribute = enables_customs_house }
			}
			every_owned_province = {
				limit = {
					has_province_modifier = GE_Customs_House
				}
				remove_province_modifier = GE_Customs_House
			}
		}
		else = {
			every_owned_province = {
				limit = {
					OR = {
						trade_goods = cloth
						trade_goods = livestock
						trade_goods = wool
					}
					NOT = {	has_province_modifier = GE_Customs_House }
				}
				add_province_modifier = {
					name = GE_Customs_House
					duration = -1
					hidden = yes
				}
			}
		}
	}

	option = {
		name = GE_Italian_Reforms_Events.4.OPT1
	}
}
